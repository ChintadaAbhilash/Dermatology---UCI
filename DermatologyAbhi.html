def StandardScaler(Xtrain , Ytrain , Xtest , Ytest):
    print(" Transforming the way you need ! ;) ")
    
    mu = Xtrain.mean(axis=0)
    std = Xtrain.std(axis=0)
    
    np.place(std, std == 0, 1)
    
    Xtrain = (Xtrain - mu) / std
    Xtest = (Xtest - mu) / std
    
    return Xtrain, Xtest , Ytrain, Ytest
